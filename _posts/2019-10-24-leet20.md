```py
# My Solution
class Solution:
    def isValid(self, s: str) -> bool:

        def check_zero_count(s):
            count_0 = s.count('()')
            count_1 = s.count('{}')
            count_2 = s.count('[]')

            if count_0+count_1+count_2!=0:
                return False
            else:
                return True

        def split(s):
            count_0 = s.count('()')
            count_1 = s.count('{}')
            count_2 = s.count('[]')


            for _ in range(count_0):
                temp_list = s.split('()')
                if '' in temp_list:
                    temp_list.remove('') 
                s = ''.join(temp_list)
            
            for _ in range(count_1):
                temp_list = s.split("{}")
                if '' in temp_list:
                    temp_list.remove('') 
                s = ''.join(temp_list)

            for _ in range(count_2):
                temp_list = s.split('[]')
                if '' in temp_list:
                    temp_list.remove('') 
                s = ''.join(temp_list)

            if not check_zero_count(s):
                s = split(s)

            return s

        s = split(s)

        if s == '':
            return True
        else:
            return False
```

```py
class Solution(object):
    def isValid(self, s):
        while "()" in s or "{}" in s or '[]' in s:
            s = s.replace("()", "").replace('{}', "").replace('[]', "")
        return s == ''
```
